static Map<String, int> computeYearCumulative(
    List<ScrobbleRecord> records, int year) {
  final monthly = computeMonthly(records);
  int cum = 0;
  final result = <String, int>{};
  for (var m = 1; m <= 12; m++) {
    final key = '$year-${m.toString().padLeft(2, '0')}';
    cum += monthly[key] ?? 0;
    result[key] = cum;
  }
  return result;
}
